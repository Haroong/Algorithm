string solution(string binary){

      switch (count) {
        case 0:
          sum += pow(2, 0);
          break;
        case 1:
          sum += pow(2, 1);
          break;
        default:
          sum += pow(2, 2);
          break;
      }
    }
    
    count++;
    
    if (count == 3 && i != 0) {
      answer += to_string(sum);
      sum = 0;
      count = 0;
    }
  }
