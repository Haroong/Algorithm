#include <iostream>
#include <stack>
#include <vector>

using namespace std;

int solution(vector<int> v, int n){
  stack<int> s;
  int answer = 0;
  
  for (int j = 0; j < n; j++) {
    if (s.empty() || v[j] < s.top()) {
      s.push(v[j]);
    } else {
      s.pop();
      while (!s.empty() && s.top() <= v[j]) {
        s.pop();
      }
      s.push(v[j]);
    }
  }
  
  answer = (int)s.size();
  cout << answer << endl;
  
  return answer;
}

int main() {
  int N, input;
  cin >> N;
  
  vector<int> v;

  return 0;
}
