n = int(input())

left_sum, right_sum = 0, 0
num_len = len(str(n))

	if digit < int(num_len / 2):
		left_sum += int(str(n)[digit])
	else:
		right_sum += int(str(n)[digit])

if left_sum == right_sum:
	print('LUCKY')
else:
	print('READY')
