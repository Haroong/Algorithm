#include <iostream>
#include <vector>
#include <string>

using namespace std;

void solution(vector<string> v){
  bool flag = false;
  
  for (int j = 0; j < 5; ++j) {
    for (int k = 0; k < v[j].size(); ++k) {
      if (v[j][k] == 'F' && v[j][k+1] == 'B' && v[j][k+2] == 'I'){
        flag = true;
        cout << j + 1 << " ";
        break;
      }
    }
  }
  
  if (flag == false) { cout << "HE GOT AWAY!" << endl; }
}

int main(){
  vector<string> v;
  string input;

  // get input
  for (int i = 0; i < 5; ++i) {
    cin >> input;
    v.push_back(input);
  }
  
  solution(v);
  
  return 0;
}
