#include <iostream>
#include <vector>
#include <string>

using namespace std;

void solution(vector<string> v, int max){
  string answer = "";
    
  for (int j = 0; j < max; j++) {
    for (int k = 0; k < 5; k++) {
      if (j >= v[k].length()){
        continue;
      } else {
        answer += v[k][j];
      }
    }
  }

  cout << answer << endl;
}

int main(){
  vector<string> v;
  string input;

  // get input
  for (int i = 0; i < 5; ++i) {
    cin >> input;
    v.push_back(input);
  }
  
  // find a max string length
  int max = (int)v[0].length();
  
  for (int k = 1; k < 5; k++) {
    if (max < v[k].length()){
      max = (int)v[k].length();
    }
  }

  solution(v, max);
  
  return 0;
}
