#include <iostream>
#include <stack>
#include <cstring>

using namespace std;

int checkOperator(char s) {
  int priority = 0;
  
  if (s == '*' || s == '/') {
    priority = 3;
  } else if (s == '+' || s == '-') {
    priority = 2;
  } else if (s == '('){
    priority = 1;
  } else {
    priority = -1;
  }
  
  return priority;
}

void toPostfix(string str) {
  string answer;
  stack<char> s;
  
  for (int i = 0; i < str.length(); ++i) {
    char ch = str[i];
    switch (ch) {
      case '+':
      case '-':
      case '*':
      case '/':
        while (!s.empty() && (checkOperator(ch) <= checkOperator(s.top()))){
          if (s.top() == '('){
            s.pop();
          } else {
            answer += s.top();
            s.pop();
          }
        }
        s.push(ch);
        break;
      case '(':
        s.push(ch);
        break;
      case ')':
        while (s.top() != '(') {
          answer += s.top();
          s.pop();
        }
        s.pop(); // '(' popí•´ì¤Œ
        break;
      default:
        // alphabet
        answer += ch;
        break;
    }
  }
  
  // ìŠ¤íƒì— ë­”ê°€ê°€ ë‚¨ì•„ìˆì„ë•Œ: ë‹¤ ë¹¼ì¤€ë‹¤
  while (!s.empty()){
    if (s.top() == '('){
      s.pop();
    } else {
      answer += s.top();
      s.pop();
    }
  }
    
  cout << answer << endl;
}

int main(){
  string input;
  cin >> input;
  
  toPostfix(input);
  
  return 0;
}
