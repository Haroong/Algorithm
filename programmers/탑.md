function solution(heights) {
  const len = heights.length;
  let answer = [],
    stack = [];

  // push elements to stack
  for (let i = 0; i < len; i++) {
    stack.push(heights[i]);
    answer.push(0);
  }

  // checking stack
  for (let j = len - 1; j > 0; j--) {
    let t = j - 1;
    for (t; t >= 0; t--) {
      if (stack[j] < stack[t]) {
        answer[j] = t + 1;
        stack.pop();
        break;
      }
    }
  }
  answer[0] = 0; // ìŠ¤íƒ ë§ˆì§€ë§‰ ì›ì†Œ
  console.log(answer);
  return answer;
}
